22  package com.xley.lfosc.impl;
23  
24  import com.xley.lfosc.IProtocol;
25  import com.xley.lfosc.IProtocolData;
26  import com.xley.lfosc.ProtocolException;
27  import com.xley.lfosc.util.LogUtil;
28  
29  import java.net.URLDecoder;
30  import java.nio.charset.Charset;
31  import java.util.*;
32  
33  public abstract class BaseProtocol implements IProtocol {
34      /**
35       * The constant resources.
36       */
37      private static final ResourceBundle resources = ResourceBundle.getBundle(BaseProtocol.class.getSimpleName(),
38              Locale.getDefault());
39  
40      @Override
41      public IProtocolData configureProtocolData(IProtocolData data) {
42          try {
43              String operation = URLDecoder.decode((String) data.getOperation(), Charset.defaultCharset().name());
44  
45              List<Object> argsList = null;
46              if (operation.contains(" ")) {
47                  String[] args = operation.substring(operation.indexOf(" ")).trim().split(" "); //find the start of the arguments
48                  operation = operation.substring(0, operation.indexOf(" ")); //find the start of the arguments
49                  argsList = new ArrayList<>();
50                  Collections.addAll(argsList, args);
51              }
52              if (argsList == null) {
53                  argsList = data.getDataList();
54              } else if (data.getDataList() != null) {
55                  argsList.addAll(data.getDataList());
56              }
57  
58              return new SimpleProtocolData(data.getType(),
59                      URLDecoder.decode((String) data.getTarget(), Charset.defaultCharset().name()),
60                      operation, argsList);
61  
62          } catch (Exception e) {
63              LogUtil.warn(getClass(), e);
64              return data;
65          }
66      }
67  
68      @Override
69      public int command(String command, String[] args) throws ProtocolException {
70          LogUtil.error(resources.getString("protocol.command.not_found"));
71          return -1;
72      }
73  }
